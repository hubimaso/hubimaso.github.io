static int32_t test_write_buffer_cb(char *buffer, void *file_data)
{
int32_t len = 0;
{
char *buf = (char *)file_data;
if (buf)
{
len += sprintf(buffer + len, "%s", buf);
Log_record(LOG_DEBUG, "%s\n", buf);
}
}
return len;
}
static int32_t test_file_name_create(void *fileio_task, int32_t thread_id, char *file_name, int32_t file_name_len)
{
if (file_name)
{
snprintf(file_name, file_name_len, "/tmp/fileio_test0%d.txt", thread_id);
return 0;
}
return -1;
}
int32_t dpdk_fileio_test_start(void)
{
int32_t ret = 0;
ret = dpdk_fileio_init((char *)"7", 1, FILE_WRITE_4K, 32);
if (ret)
{
return FAILED;
}
FILEIO_TASK_INPUT task_input;
memset(&task_input, 0, sizeof(FILEIO_TASK_INPUT));
task_input.file_blank = 0;
task_input.file_compress = 0;
task_input.file_delimiter = '|',
task_input.fileio_type = FILEIO_ASYNC;
task_input.filename_create_cb = test_file_name_create;
strcpy(task_input.task_name, "fio_test");
task_input.timer_period = 1000;
task_input.write_buffer_cb_num = 1;
task_input.write_buffer_cb[0] = test_write_buffer_cb;
task_input.write_period_limit = 5;
task_input.write_size_limit = 20 * 1024 * 1024;//20M
void *task_handler = dpdk_fileio_task_create(&task_input);
char buf[64] = "1234567890qwertyuiopasdfghjklzxcvbnm";
ret = dpdk_fileio_task_add(task_handler, (void *)buf, 0);
if (ret)
{
return FAILED;
}
return SUCCESS;
}
