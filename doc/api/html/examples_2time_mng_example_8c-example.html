#ifdef __cplusplus
extern "C" {
#endif
#include "libnap.h"
int32_t time_test_start(void)
{
int32_t ret;
/* 时间格式转换 */
char time_buf[64] = {'\0'};
/* 按照 20181108170723 格式将系统时间转换成时间戳*/
ret = get_timestamp_covert_tostring(time_buf, 64, FORMAT_NOSPACE, 0);
if (ret) return FAILED;
/* 按照 2018-11-08 17:07:23 格式将系统时间转换成时间戳*/
ret = get_timestamp_covert_tostring(time_buf, 64, FORMAT_WITHBAR, 0);
if (ret) return FAILED;
/* 按照 20181108170723 格式将指定时间转换成时间戳*/
ret = get_timestamp_covert_tostring(time_buf, 64, FORMAT_NOSPACE, 1545639201);
if (ret) return FAILED;
/* 按照 2018-11-08 17:07:23 格式将指定时间转换成时间戳*/
ret = get_timestamp_covert_tostring(time_buf, 64, FORMAT_WITHBAR, 1545639201);
if (ret) return FAILED;
uint64_t timestamp = 0;
/* 按照 2018-11-08 17:07:23 格式将指定时间转换成时间戳 */
ret = get_timetring_covert_tostamp("2018-11-08 17:07:23", FORMAT_WITHBAR, &timestamp);
if (ret) return FAILED;
/* 按照 20181108170723 格式将指定时间转换成时间戳 */
ret = get_timetring_covert_tostamp("20181108170723", FORMAT_NOSPACE, &timestamp);
if (ret) return FAILED;
return SUCCESS;
}
#ifdef __cplusplus
}
#endif
