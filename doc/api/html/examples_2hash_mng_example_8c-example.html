#ifdef __cplusplus
extern "C" {
#endif
#include "libnap.h"
typedef struct _ST_HASH_TEST_NODE
{
char url[64]; // 如果url为动态内存分配，则需要使用者释放
int32_t id;
}HASH_TEST_NODE;
int32_t ut_hash_print_cb(void *hash_value, void *arg)
{
HASH_TEST_NODE *data = hash_value;
Log_record(LOG_INFO, "find node: id = %d, url = %s, addr = %p\n", data->id, data->url, data);
return SUCCESS;
}
int32_t ut_hash_test_start(void)
{
/* 创建UThash表 */
UT_HASH_INPUT hash_input;
strncpy(hash_input.hash_name,"ut_hash_test", strlen("ut_hash_test"));
hash_input.cache_size = RTE_CACHE_LINE_SIZE;
hash_input.data_free = NULL;
hash_input.data_size = sizeof(HASH_TEST_NODE);
hash_input.hash_size = 1024;
hash_input.hash_type = UT_HASH_STR;
hash_input.numa_id = 0;
void *ut_hash_handler = NULL;
ut_hash_handler = ut_hash_create(&hash_input);
if (unlikely(ut_hash_handler == NULL)) return FAILED;
/* 因为hash_size不为0，value内存从平台中申请 */
UT_HASH_NODE *node = ut_hash_get_memnode(ut_hash_handler);
if (unlikely(node == NULL)) return FAILED;
HASH_TEST_NODE *value = node->hash_value;
value->id = 0;
strncpy(value->url, "http://192.168.1.100:1024/examples.html", strlen("http://192.168.1.100:1024/examples.html"));
/* 新增key value */
int32_t ret = ut_hash_add(ut_hash_handler, value->url, node);
if (unlikely(ret == FAILED)) return FAILED;
/* 查找key value */
UT_HASH_NODE *find_node = ut_hash_find(ut_hash_handler, "http://192.168.1.100:1024/examples.html");
if (unlikely(find_node == NULL)) return FAILED;
ut_hash_for_each_cb(ut_hash_handler, ut_hash_print_cb, NULL);
/* 删除key value */
ret = ut_hash_delete(ut_hash_handler, find_node);
if (unlikely(ret == FAILED)) return FAILED;
ut_hash_destroy(ut_hash_handler);
return SUCCESS;
}
